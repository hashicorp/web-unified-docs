Use the [`vault_plugin_pinned_version`](https://registry.terraform.io/providers/hashicorp/vault/latest/docs/resources/plugin_pinned_version)
resource to pin the plugin version, [`vault_mount`](https://registry.terraform.io/providers/hashicorp/vault/latest/docs/resources/mount)
to enable the secrets engine, and [`vault_os_secret_backend`](https://registry.terraform.io/providers/hashicorp/vault/latest/docs/resources/os_secret_backend)
to configure the backend:

```hcl
resource "vault_plugin" "os" {
  type    = "secret"
  name    = "vault-plugin-secrets-os"
  command = "vault-plugin-secrets-os"
  version = "v0.1.0+ent" # https://releases.hashicorp.com/vault-plugin-secrets-os
}

resource "vault_plugin_pinned_version" "os" {
  type    = vault_plugin.os.type
  name    = vault_plugin.os.name
  version = vault_plugin.os.version
}

resource "vault_mount" "os" {
  path = "os"
  type = vault_plugin_pinned_version.os.name
}

resource "vault_os_secret_backend" "os" {
  mount = vault_mount.os.path
}
```
