Use the [`vault_os_secret_backend_account`](https://registry.terraform.io/providers/hashicorp/vault/latest/docs/resources/os_secret_backend_account)
resource to configure both accounts:

```hcl
resource "vault_os_secret_backend_account" "mgmt_user" {
  mount       = vault_os_secret_backend.os.mount
  host        = vault_os_secret_backend_host.my_box.name
  name        = "mgmt-user"
  username    = "super-user"
  password_wo = "super-secret"
}

resource "vault_os_secret_backend_account" "app_user" {
  mount               = vault_os_secret_backend.os.mount
  host                = vault_os_secret_backend_host.my_box.name
  name                = "app-user"
  username            = "user-1"
  password_wo         = "bigsecret"
  parent_account_ref  = vault_os_secret_backend_account.mgmt_user.name
}
```
